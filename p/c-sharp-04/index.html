class AnimalComparer : IComparer<Animal>
{
    int IComparer<Animal>.Compare(Animal a, Animal b)
    {
        if (a == null) return b == null ? 0 : -1;
        return b == null ? 1 : a.Age.CompareTo(b.Age);
    }
}

    }
}

    Console.Write(node.Value + " "); // 1, 2
    node = node.Next;
}
