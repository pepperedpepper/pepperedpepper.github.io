void FloydWarshall(vector<vector<int>>& graph, int nodeNum)
{
    for (int k = 1; k < nodeNum + 1; k++)
        for (int a = 1; a < nodeNum + 1; a++)
            for (int b = 1; b < nodeNum + 1; b++)
                graph[a][b] = min(graph[a][k] + graph[k][b], graph[a][b]);
}

            int curF = -pq.top().f;
            pq.pop();

            // ëª©ì ì§€ì— ë„ë‹¬í–ˆë‹¤ë©´ ëë‚´ê¸°
            if (cur.x == des.x && cur.y == des.y) break;
                
            // ê¸°ì¡´ì˜ Fê°’ë³´ë‹¤ < í˜„ì¬ ë…¸ë“œì˜ Fê°’ì´ ë” í¬ë©´ (ë” ì¢‹ì€(ì§§ì€) ê±°ë¦¬ê°’ì´ ì•„ë‹ˆë©´)
            // í•„ìš” ì—†ë‹¤
            if (data[cur.x][cur.y].f < curF) continue;

            for (int i = 0; i < DIR_LENGTH; ++i)
            {
                Pos next = { cur.x + DIR[i].x, cur.y + DIR[i].y };

                if (next.x < 0 || next.y < 0) continue;
                if (row <= next.x || col <= next.y) continue;
                if (map[next.x][next.y] == INF) continue;
                
                // ë…¸ë“œ nextì˜ Gê°’ = ì´ì „ë…¸ë“œ curì˜ Gê°’ + ì´ì „ë…¸ë“œì™€ nextì˜ ê±°ë¦¬
                // ë…¸ë“œ nextì˜ Fê°’ = Gê°’ + ëª©ì ì§€ë…¸ë“œì™€ nextì˜ ê±°ë¦¬(H)
                int nextG = data[cur.x][cur.y].g + getDistance(cur, next);
                int nextH = getDistance(next, des);
                int nextF = nextG + nextH;

                if (data[next.x][next.y].f < nextF) continue;
                
                data[next.x][next.y].g = nextG;
                data[next.x][next.y].h = nextH;
                data[next.x][next.y].f = nextG + nextH;
                data[next.x][next.y].prev = cur;
                pq.push({ -data[next.x][next.y].f, next });
            }
        }
        
        printTable(data, row, col, start, des);
        printPath(data, start, des);
    }
    
    void printTable(vector<vector<NodeData>>& data, int row, int col, Pos start, Pos des)
    {
        printf("ì‹œì‘ ë…¸ë“œ : {%d, %d}, ëª©ì ì§€ ë…¸ë“œ : {%d, %d}\n", start.x, start.y, des.x, des.y);
        printf("í˜•ì‹ : G + H = F (ì´ì „ë…¸ë“œx, ì´ì „ë…¸ë“œy)\n");
        
        for (int i = 0; i < row; i++)
        {
            for (int j = 0; j < col; j++)
            {
                if (data[i][j].g != INF)
                {
                    printf("%2d+%2d=%2d(%d, %d) ",
                       data[i][j].g, data[i][j].h, data[i][j].f,

                }
                else
                {
                    cout << setw(15) << "";
                }
            }
            cout << endl;
        }
        cout << endl;
    }
    
    void printPath(vector<vector<NodeData>>& data, Pos start, Pos des)
    {
        // ê²½ë¡œ ì¶œë ¥ì„ ìœ„í•´ ëª©ì ì§€ë¶€í„° ë˜ëŒì•„ê°€ë©° pathì— ê²½ë¡œ ë„£ê¸°
        vector<Pos> path;
        
        path.push_back(des);
        Pos cur = data[des.x][des.y].prev;
        
        while (1)
        {
            path.push_back(cur);
            if (cur.x == start.x && cur.y == start.y) break;
                
            cur = data[cur.x][cur.y].prev;
        }
        
        cout << "ì´ë™ê²½ë¡œ : ";

        for (int i = path.size() - 1; i >= 0; i--)
        {
            cout << "{" << path[i].x << ", " << path[i].y << "}";
            
            if (i != 0) cout << " > ";
        }
        
        cout << endl;
    }

struct EdgeData
{
    int a;
    int b;
    int cost;
    
    // ë¹„ìš©ì´ ì‘ì€ ê²ƒì´ ì•ìœ¼ë¡œ ê°„ë‹¤. 
    bool operator<(EdgeData b)
    {
        return cost < b.cost;
    }

void UnionParent(int a, int b, vector<int>& parent)
{
    a = FindParent(a, parent);
    b = FindParent(b, parent);

    if (a > b) parent[a] = parent[b];
    else       parent[b] = parent[a];
}

int FindParent(int idx, vector<int>& parent)
{
    if (idx != parent[idx])
        parent[idx] = FindParent(parent[idx], parent);

    return parent[idx];
}
