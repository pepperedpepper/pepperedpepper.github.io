void InsertionSort(int* arr, int size)
{
    for (int i = 1; i < size; i++)   // i = 1 ~ size-1
    {
        for (int j = i; j >= 1; j--) // j = i ~ 1 
        {
            if (arr[j] < arr[j - 1]) 
                Swap(arr[j], arr[j - 1]);

                break;
        }
    }
}

void SelectionSort(int* arr, int size)
{
    int min = 0;
    for (int i = 0; i < size; i++)
    {
        min = i;

        for (int j = i + 1; j < size; j++)
            if (arr[min] > arr[j]) 
                min = j;

        Swap(arr[i], arr[min]);
    }
}

void BubbleSort(int* arr, int size)
{
    for (int i = 1; i < size; i++)
        for (int j = 0; j < size - i; j++)
            if (arr[j] > arr[j + 1])
                Swap(arr[j], arr[j + 1]);
}

void InsertionSort(int* arr, int first, int last, int gap)
{
    for (int i = first + gap; i <= last; i += gap) 
    {
        for (int j = i - gap; j >= first; j -= gap) 
        {
            if (arr[j] > arr[i])
                Swap(arr[j], arr[i]);
            else 
                break;
        }
    }
}

void QuickSort(int* arr, int firstIdx, int lastIdx)
{
    if (firstIdx >= lastIdx) return;

    int piviotIdx = firstIdx;
    int largeIdx = firstIdx + 1;
    int smallIdx = lastIdx;

    // ÎåÄÎûµÏ†ÅÏúºÎ°ú ÏöîÏÜåÏùò Í∞úÏàò n ÎßåÌÅº Î∞òÎ≥µÌïòÎ©¥ÏÑú Í≤ÄÏÇ¨ÌïòÍ≤å ÎêúÎã§. 
    while(largeIdx <= smallIdx) // Í≤πÏπòÎ©¥, ÎÇòÍ∞ÑÎã§
    {
        // >>> ÌÅ∞ Í∞íÏùÑ Ï∞æÏïÑÏÑú.
        // ÎßàÏßÄÎßâ Ïù¥ÌïòÏù¥Í≥†, Ïù¥Î≤à Í∞íÏù¥ ÌîºÎ≤óÎ≥¥Îã§ ÏûëÏùÑ Îïå Î∞òÎ≥µÌïúÎã§. 
        // (Ï†êÏ†ê Ïª§ÏßÄÎ©¥ Î∞∞Ïó¥ÏùÑ ÎÑòÏñ¥ÏÑ†Îã§... ÌïòÏßÄÎßå Í∑∏ ÎßêÏùÄ Í≤πÏ≥§Îã§Îäî ÎúªÏù¥Î©∞, Í≤πÏ≥§ÏùÑ Îïå Ïù¥ Í∞íÏùÑ Ïì∞ÏßÑ ÏïäÏúºÎØÄÎ°ú ÏÉÅÍ¥ÄÏóÜÎã§)
        while (largetIdx <= lastIdx && arr[largetIdx] < arr[piviotIdx])
            largetIdx++;

        // <<< ÏûëÏùÄ Í∞íÏùÑ Ï∞æÏïÑÏÑú.
        // Ï≤´Î≤àÏß∏ Ïù¥ÏÉÅÏù¥Í≥†, Ïù¥Î≤à Í∞íÏù¥ ÌîºÎ≤óÎ≥¥Îã§ ÌÅ¥ Îïå Î∞òÎ≥µÌïúÎã§.
        // (Ï†êÏ†ê ÏûëÏïÑÏßÄÎ©¥ ÌîºÎ≤ó Ïù¥ÌõÑ Îëê Î≤àÏß∏ Ïπ∏ÍπåÏßÄ Í∞ÑÎã§)
        while (smallIdx >= firstIdx && arr[smallIdx] > arr[piviotIdx])
            smallIdx--;

        // Í≤πÏπòÎ©¥, ÌîºÎ≤óÍ≥º ÏûëÏùÄ Í≤ÉÏùÑ ÍµêÏ≤¥ÌïúÎã§, Í∑∏Î¶¨Í≥† ÎÇòÍ∞ÄÍ≤å ÎêúÎã§
        // Í≤πÏπúÍ≤å ÏïÑÎãàÎ©¥, ÎëòÏù¥ ÍµêÏ≤¥ÌïúÎã§
        if (largetIdx > smallIdx) 
            Swap(arr[smallIdx], arr[piviotIdx]);
        else
            Swap(arr[smallIdx], arr[largetIdx]);  
    }

    // Í≤πÏπ† ÎïåÍπåÏßÄ ÍµêÌôòÌïú Í≤∞Í≥º, Í≤πÏπú Î∂ÄÎ∂ÑÏùÑ Ï§ëÏã¨ÏúºÎ°ú ÏôºÏ™ΩÏùÄ ÏûëÏùå Ïò§Î•∏Ï™ΩÏùÄ ÌÅºÏúºÎ°ú ÎÇòÎâòÍ≤å ÎêúÎã§
    // Í≤πÏ≥§ÏùÑ ÎïåÎäî ÏûëÏùÄ Í≤ÉÍ≥º ÍµêÌôòÌñàÏúºÎØÄÎ°ú ÏûëÏùÄ Í≤ÉÏù¥ Ï§ëÏã¨Ïù¥ ÎêúÎã§.
    // Í≥ÑÏÜçÌï¥ÏÑú Ï†àÎ∞òÏî© ÎÇòÎàÑÍ≤å ÎêúÎã§Î©¥ Í∑∏ ÎÜíÏù¥Îäî log nÏùº Í≤ÉÏù¥Îã§. 
    QuickSort(arr, firstIdx, smallIdx - 1); 
    QuickSort(arr, smallIdx + 1, lastIdx);
}

void makeHeap(int* arr, int start, int last)
{
    // Î£®Ìä∏ÏóêÏÑú ÏãúÏûëÌï¥ÏÑú
    int rootToMove = arr[start];
    int parentIdx;
    int maxChildIdx = start;

    for (parentIdx = start; parentIdx < (last + 1) / 2; parentIdx = maxChildIdx)
    {
        int leftChildIdx = (parentIdx * 2) + 1;
        int rightChildIdx = (parentIdx * 2) + 2;

        // ÏôºÏ™ΩÍ≥º Ïò§Î•∏Ï™Ω ÏûêÏãù Ï§ëÏóê Î£®Ìä∏Î≥¥Îã§ ÌÅ∞ Í∞íÏù¥ ÏûàÏúºÎ©¥
        maxChildIdx = (rightChildIdx <= last && arr[rightChildIdx] > arr[leftChildIdx]) ? rightChildIdx : leftChildIdx;

        if (rootToMove >= arr[maxChildIdx])
            break;

        // Í∑∏ ÌÅ∞ ÏûêÏãùÏù¥ Î∂ÄÎ™®Ïùò ÏûêÎ¶¨Î•º Íø∞Ï∞¨Îã§.
        arr[parentIdx] = arr[maxChildIdx];
        
        // Í∑∏ ÏûêÏãùÏùÑ Î£®Ìä∏Î°ú Î≥¥Í≥† Îã§Ïãú Î∞òÎ≥µÌïúÎã§.
    }

    // ÎçîÏù¥ÏÉÅ ÌÅ∞ ÏûêÏãùÏù¥ ÏóÜÏùÑ Îïå, Í∑∏ ÏûêÎ¶¨Í∞Ä Ï†ÅÏ†àÌïú ÏûêÎ¶¨Ïù¥Îã§.
    arr[parentIdx] = rootToMove;
}

void heapSort(int* arr, int size)
{
    // Ï†úÏùº ÏûëÏùÄ Ïù¥ÏßÑ Ìä∏Î¶¨Ïùò Î£®Ìä∏ÏóêÏÑú Î∂ÄÌÑ∞ Ìïú Î∂ÄÎ™®Ïî© Ïò¨ÎùºÍ∞ÄÎ©¥ÏÑú Ï†ÑÏ≤¥Î•º ÌûôÏúºÎ°ú ÎßåÎì†Îã§.
    int lastRootIndex = size / 2 - 1;
    for (int i = lastRootIndex; i >= 0; i--)
        makeHeap(arr, i, size - 1);

    // Ìïú ÏöîÏÜåÏî© Ìä∏Î¶¨ÏóêÏÑú ÎπºÎ©¥ÏÑú Î∞òÎ≥µÌïòÎØÄÎ°ú n ÎßåÌÅº Î∞òÎ≥µÎêúÎã§.
    for (int i = 1; i < size; i++)
    {
        // Ïù¥ÏßÑ Ìä∏Î¶¨Ïùò Î£®Ìä∏(0)Î•º Îß® Îí∑ ÏöîÏÜå(size - 1)ÏôÄ Î∞îÍøîÏÑú Îß® Îí§Îäî Ï†úÏùº ÌÅ∞ Î£®Ìä∏Í∞Ä Ï∞®Í≥°Ï∞®Í≥° Ï†ÄÏû•ÎêòÍ≤å ÌïúÎã§.
        swap(arr[0], arr[size - i]);
        
        // Ïù¥Ï†ú Îß® ÎÅùÏùÄ ÌÅ∞Í≤å Ï†ÄÏû•ÎêòÏóàÏúºÎØÄÎ°ú Î£®Ìä∏Î∂ÄÌÑ∞ Í∑∏ Ï†Ñ(size - 2)ÎÖ∏ÎìúÍπåÏßÄ Îã§Ïãú ÌûôÏùÑ ÎßåÎì†Îã§.
        makeHeap(arr, 0, size - i - 1);

void swap(int& a, int& b)
{

    a = b;
    b = temp;

void MergeSort(int* arr, int first, int last)
{
    int size = last - first + 1;
    if (size <= 1) return;

    int mid = (first + last) / 2;

    // Í≥ÑÏÜç Ï†àÎ∞òÏî© ÎÇòÎàÑÎØÄÎ°ú Ï¥ù log nÎ≤à Î∞òÎ≥µÎêúÎã§. 
    MergeSort(arr, first, mid);

    // Ìï©ÏπúÎã§.
    Merge(arr, first, last);
}

void Merge(int* arr, int first, int last)
{
    int mid = (first + last) / 2;
    int size = (last - first) + 1;

    int* sorted = new int[size]();

    int idx = 0;
    int left = first;
    int right = mid + 1;

    // Ï¥ù ÏöîÏÜåÏùò Í∞úÏàòÎßåÌÅº nÎ≤à Î∞òÎ≥µÌïòÎ©¥ÏÑú ÎπÑÍµêÍ∞Ä ÎêúÎã§. 

    // Ïù∏Îç±Ïä§Î•º ÎçîÌï¥Í∞ÄÎ©¥ÏÑú ÎπÑÍµêÌïúÎã§.
    while (left <= mid && right <= last) 
    {
        if (arr[left] <= arr[right])       sorted[idx++] = arr[left++];
        else if (arr[left] > arr[right])   sorted[idx++] = arr[right++];
    }

    // ÎÇ®ÏùÄ Î∞∞Ïó¥ÏùÄ Î™®Îëê ÏàúÏÑúÎåÄÎ°ú Ï∂îÍ∞ÄÌï¥Ï§ÄÎã§. 
    if (left > mid)
    {
        for (int i = right; i <= last; i++)
            sorted[idx++] = arr[i];
    }
    else if (right > last)
    {
        for (int i = left; i <= mid; i++)
            sorted[idx++] = arr[i];
    }

    // ÏõêÎûò Î∞∞Ïó¥Î°ú ÏòÆÍ∏¥Îã§. 
    for (int i = 0; i < size; i++)
        arr[first + i] = sorted[i];
}

void BinarySearch_Loop(int* arr, int size, int target)
{
    int start = 0;
    int end = size - 1;
    int mid = (end + start) / 2;

    while(start < end)
    {
        if (arr[mid] == target)
        {
            cout << "Found it in index : " << mid << endl;
            return;
        }
        else if (arr[mid] < target) start = mid + 1;
        else                        end = mid - 1;

        mid = (end + start) / 2;
    }

    cout << "Couldn't find it." << endl;
}

void BinarySearch_Recursion(int* arr, int start, int end, int target)
{
    if (start > end) return;

    int mid = (start + end) / 2;

    if (arr[mid] == target)
    {
        cout << "Found it in index : " << mid << endl;
        return;
    }
    else if (arr[mid] < target)
    {
        BinarySearch_Recursion(arr, mid + 1, end, target);
        start = mid + 1;
    }
    else
    {
        BinarySearch_Recursion(arr, start, mid - 1, target);
    }
}
